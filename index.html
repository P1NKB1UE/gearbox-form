2
In the sheet → Extensions → Apps Script → delete all existing code → paste this:
function doPost(e){
  var s=SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Jobs");
  var d=JSON.parse(e.postData.contents);
  var today=Utilities.formatDate(new Date(),"Asia/Kuala_Lumpur","yyMMdd");
  var last=s.getLastRow();
  var count=0;
  if(last>1){
    var ids=s.getRange(2,1,last-1,1).getValues().flat();
    count=ids.filter(function(id){return String(id).indexOf("SF-"+today)===0;}).length;
  }
  var jobId="SF-"+today+"-"+String(count+1).padStart(3,"0");
  s.appendRow([jobId,d.date,d.car,d.year,d.plate,d.mileage,d.gearboxType,d.when,d.symptoms,d.jerkGears,d.slipGears,d.delayGears,d.noEngagePos,d.noiseDetails,d.remarks,d.advisor,"Pending"]);
  return ContentService.createTextOutput(JSON.stringify({status:"ok",jobId:jobId})).setMimeType(ContentService.MimeType.JSON);
}
